def checkRematch(winner, loser):
    DB, c = connect()

    query = "SELECT player1, player2 from Matches;"
    c.execute(query)
    res = c.fetchall()

    for i in res:
        if (winner == i[0] or winner == i[1])
        and (loser == i[0] or loser == i[1]):
            raise RuntimeError("Cannot play rematch")
    DB.close()
